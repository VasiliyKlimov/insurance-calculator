function calculatePremium() { const program = document.querySelector('input[name="program"]:checked').value; const insuranceDays = parseInt(document.getElementById('insuranceDays').value, 10); const coverageAmount = parseFloat(document.getElementById('coverageAmount').value, 10); const clientAge = parseInt(document.getElementById('clientAge').value, 10); const region = document.getElementById('region').value; const hobby = document.getElementById('hobby').value; const diseaseHistory = document.getElementById('diseaseHistory').value; const disabilityCompensation = parseFloat(document.getElementById('disabilityCompensation').value, 10); const healthStatus = document.getElementById('healthStatus').value; const franchise = parseFloat(document.getElementById('franchise').value, 10); const additionalOption = document.getElementById('additionalOption').value; if (isNaN(insuranceDays) || insuranceDays <= 0 || isNaN(coverageAmount) || coverageAmount <= 0) { document.getElementById('premiumResult').textContent = 'Пожалуйста, введите корректные данные для расчёта.'; return; } console.log("Возраст клиента:", clientAge); console.log("Регион:", region); console.log("Занятие или хобби:", hobby); console.log("История заболеваний:", diseaseHistory); console.log("Сумма компенсации за потерю трудоспособности:", disabilityCompensation); console.log("Состояние здоровья:", healthStatus); console.log("Франшиза:", franchise); console.log("Дополнительная опция:", additionalOption); let baseRate; switch (program) { case 'basic': baseRate = 0.01; break; case 'complex': baseRate = 0.015; break; case 'complex_ambulance': baseRate = 0.02; break; default: baseRate = 0.01; } // Учет дополнительных факторов let riskFactor = 1.0; // Учет хобби switch (hobby) { case 'low': riskFactor *= 1.0; break; case 'medium': riskFactor *= 1.2; break; case 'high': riskFactor *= 1.5; break; } // Учет истории заболеваний switch (diseaseHistory) { case 'none': riskFactor *= 1.0; break; case 'minor': riskFactor *= 1.1; break; case 'serious': riskFactor *= 1.3; break; } // Учет состояния здоровья switch (healthStatus) { case 'excellent': riskFactor *= 1.0; break; case 'good': riskFactor *= 1.1; break; case 'fair': riskFactor *= 1.2; break; case 'poor': riskFactor *= 1.5; break; } // Учет франшизы let franchiseDiscount = 0; if (franchise > 0) { franchiseDiscount = franchise * 0.1; // Скидка 10% от суммы франшизы } // Учет дополнительной опции let additionalOptionCost = 0; switch (additionalOption) { case 'homeCare': additionalOptionCost = 5000; // Дополнительная стоимость break; case 'rehabilitation': additionalOptionCost = 10000; // Дополнительная стоимость break; } // Расчет премии let premium = (coverageAmount * baseRate * insuranceDays / 365) * riskFactor; premium -= franchiseDiscount; // Применение скидки за франшизу premium += additionalOptionCost; // Добавление стоимости дополнительной опции // Учет компенсации за потерю трудоспособности if (disabilityCompensation > 0) { premium += disabilityCompensation * 0.05; // Дополнительная стоимость 5% от суммы компенсации } document.getElementById('premiumResult').textContent = `Расчётная стоимость страхования: ${premium.toFixed(2)} ₽`; }
